#include <iostream>
#include <fstream>
#include <string>
#include "BigInteger.h"

using namespace std;

int main(int argc, char* argv[]) {
    if (argc != 3) {
        cerr << "Usage: " << argv[0] << " <input file> <output file>" << endl;
        return EXIT_FAILURE;
    }

    ifstream inFile(argv[1]);
    if (!inFile.is_open()) {
        cerr << "Error: Could not open " << argv[1] << " for reading." << endl;
        return EXIT_FAILURE;
    }

    ofstream outFile(argv[2]);
    if (!outFile.is_open()) {


        cerr << "Error: Could not open " << argv[2] << " for writing." << endl;
        return EXIT_FAILURE;
    }

    string strA, strB;
    inFile >> strA >> strB;

    BigInteger A(strA), B(strB);
    const BigInteger three("3"), two("2"), nine("9"), sixteen("16");

    outFile << A << '\n' << '\n'
            << B << '\n' << '\n'
            << (A + B) << '\n' << '\n'
            << (A - B) << '\n' << '\n'
            << (A - A) << '\n' << '\n'
            << (A.mult(three) - B.mult(two)) << '\n' << '\n'
            << (A * B) << '\n' << '\n'
            << (A * A) << '\n' << '\n'
            << (B * B) << '\n' << '\n'
            << (nine * A * A * A * A + sixteen * B * B * B * B * B) << '\n';

    inFile.close();
    outFile.close();

    return EXIT_SUCCESS;
}

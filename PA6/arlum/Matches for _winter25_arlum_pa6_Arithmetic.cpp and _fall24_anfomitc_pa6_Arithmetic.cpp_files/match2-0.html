#include <fstream>
#include "BigInteger.h"

using namespace std;

int main(int argc, char *argv[]) {
    // if not given enough arguments (the input and output file) we fail program
    // referenced from FileIO.cpp from professors example code
    if (argc != 3) {


        cerr << "Usage: " << argv[0] << " <input file> <output file>" << endl;
        exit(EXIT_FAILURE);
    }

    // open files for reading and writing - referenced from FileIO.cpp from professors example code
    ifstream in; 
    ofstream out; 
    in.open(argv[1]);
    if( !in.is_open() ){
        cerr << "Unable to open file " << argv[1] << " for reading" << endl;
        return(EXIT_FAILURE);
    }

    out.open(argv[2]);
    if( !out.is_open() ){


        cerr << "Unable to open file " << argv[2] << " for writing" << endl;
        return(EXIT_FAILURE);
    }
    
    // get the information from the files
    string line; 

    // read first bigint 
    getline(in,line);
    BigInteger A(line);
    
    // read empty line and ignore
    getline(in,line);

    // read second big int 
    getline(in,line);
    BigInteger B(line);

    // format output:
    // A, B
    out << A << endl;
    out<<endl;
    out << B << endl;
    out<<endl;

    // A+B
    out << A+B << endl;
    out<<endl;

    // A-B 


    out << A-B << endl;
    out<<endl;

    // A-A
    out << A-A << endl;
    out<<endl;

    // 3A-2B
    BigInteger three = BigInteger("3");
    BigInteger two = BigInteger("2"); 
    out << three*A - two*B << endl;
    out<<endl;

    // AB
    out << A*B << endl;
    out<<endl;

    // A^2
    out << A*A << endl;
    out<<endl;

    // B^2
    out << B*B << endl;
    out<<endl;

    // 9A^4 + 16B^5
    BigInteger nine = BigInteger("9");
    BigInteger sixteen = BigInteger("16"); 
    out << nine*(A*A*A*A) + sixteen*(B*B*B*B*B) << endl;
    out<<endl; 
    //out<<endl;

    // close files 
    in.close(); 
    out.close(); 

    return 0; 
}

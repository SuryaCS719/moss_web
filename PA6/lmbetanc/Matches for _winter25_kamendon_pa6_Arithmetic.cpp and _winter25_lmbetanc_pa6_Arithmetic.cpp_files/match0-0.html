#include <iostream>
#include <fstream>
#include <string>
#include "BigInteger.h"
#include "List.h"



using namespace std;

// main function
int main(int argc, char* argv[]){
    if(argc != 3){
        std::cout << "Invalid number of arguments" << std::endl;
        return 1;
    }

    std::ifstream inputFile(argv[1]);
    std::ofstream outputFile(argv[2]);

    if(!inputFile){
        std::cout << "Error opening input file" << std::endl;
        return 1;
    }
    if(!outputFile){
        std::cout << "Error opening output file" << std::endl;
        return 1;
    }

    std::string firstLine, secondLine;
    inputFile >> firstLine >> secondLine;

    BigInteger numA(firstLine);
    BigInteger numB(secondLine);

    outputFile << numA << endl << endl;
    outputFile << numB << endl << endl;

    outputFile << (numA + numB) << endl << endl;


    outputFile << (numA - numB) << endl << endl;
    outputFile << (numA - numA) << endl << endl;

    BigInteger three(3);
    BigInteger two(2);
    outputFile << ((numA * three) - (numB * two)) << endl << endl;

    outputFile << (numA * numB) << endl << endl;
    outputFile << (numA * numA) << endl << endl;
    outputFile << (numB * numB) << endl << endl;

    BigInteger nine(9);
    BigInteger sixteen(16);
    BigInteger A4 = numA * numA * numA * numA;
    BigInteger B5 = numB * numB * numB * numB * numB;
    outputFile << ((A4 * nine) + (B5 * sixteen)) << endl << endl;

    inputFile.close();
    outputFile.close();
    return 0;
}

// Arithmetic.cpp
// Client function that uses both BigInteger and List ADT to implement 
// arithmetic operations. This will read from an input file contatining
// 3 lines.
// Line 1 will take an optional sign character (1 or -) follewed by # of
// decimal digits. Line 2 is blank. Line 3 will take an optional sign
// chracter (+ or -) follewed by a # of decimal digits. 
//-----------------------------------------------------------------------------

#include <iostream>
#include <fstream>
#include <string>
#include "BigInteger.h"
#include "List.h"



using namespace std;

// main function
int main(int argc, char* argv[]){
    if(argc != 3){
        std::cout << "Invalid number of arguments" << std::endl;
        return 1;
    }

    std::ifstream in(argv[1]); // Open input file
    std::ofstream out(argv[2]); // Open output file

    if(!in){
        std::cout << "Error opening input file" << std::endl;
        return 1;
    }
    if(!out){
        std::cout << "Error opening output file" << std::endl;
        return 1;
    }

    std::string line1, line2, line3;

    // Read first number (A)
    getline(in, line1);

    // Read blank line
    getline(in, line2);

    // Read second number (B)
    getline(in, line3);

    // Create BigInteger objects
    BigInteger A(line1);
    BigInteger B(line3);
    
    out << A << endl << endl;
    out << B << endl << endl;


    out<< (A + B) << endl << endl;


    out << (A - B) << endl << endl;
    out << (A - A) << endl << endl;
    
    BigInteger three(3);
    BigInteger two(2);
    out << ((A * three) - (B * two)) << endl << endl;

    out << (A * B) << endl << endl;
    out << (A * A) << endl << endl;
    out << (B * B) << endl << endl;

    BigInteger nine(9);
    BigInteger sixteen(16);
    BigInteger A4 = A * A * A * A;
    BigInteger B5 = B * B * B * B * B;
    out << ((A4 * nine) + (B5 * sixteen)) << endl << endl;
 
    
    in.close();
    out.close();
    return 0;
}

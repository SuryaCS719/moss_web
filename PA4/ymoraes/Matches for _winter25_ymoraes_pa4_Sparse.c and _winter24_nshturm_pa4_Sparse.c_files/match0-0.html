#include <stdio.h>
#include <stdlib.h>
#include "Matrix.h"



int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Usage: %s <input file> <output file>\n", argv[0]);
        return 1;
    }

    const char* fIn = argv[1];
    FILE* in = fopen(fIn, "r");
    if (in == NULL) {
        perror("error opening file");
        return 1;
    }

    const char* fOut = argv[2];
    FILE* out = fopen(fOut, "w");
    if (out == NULL) {
        perror("error opening file");
        return 1;
    }


    int n, a, b;

    fscanf(in, "%d %d %d\n", &n, &a, &b);

    Matrix Matrix1 = newMatrix(n);
    Matrix Matrix2 = newMatrix(n);

    int row, col;
    double value;

    for (int i = 0; i < a; i++) {
        fscanf(in, "%d %d %lf\n",&row, &col, &value);
        changeEntry(Matrix1, row, col, value);
    }

    for (int j = 0; j < b; j++) {
        fscanf(in, "%d %d %lf\n", &row, &col, &value);


        changeEntry(Matrix2, row, col, value);
    }

    fclose(in);

    fprintf(out, "A has %d non-zero entries: \n", NNZ(Matrix1));
    printMatrix(out, Matrix1);

    printf("\n");

    fprintf(out, "B has %d non-zero entries: \n", NNZ(Matrix2));
    printMatrix(out, Matrix2);

    printf("\n");

    Matrix scalr = scalarMult(1.5, Matrix1);
    fprintf(out, "(1.5)*A =\n");
    printMatrix(out, scalr);

    printf("\n");

    Matrix sum1 = sum(Matrix1, Matrix2);
    fprintf(out, "A+B = \n");
    printMatrix(out, sum1);

    printf("\n");

    Matrix sum2 = sum(Matrix1, Matrix1);
    fprintf(out, "A+A = \n");
    printMatrix(out, sum2);

    printf("\n");

    Matrix diff1 = diff(Matrix2, Matrix1);
    fprintf(out, "B-A = \n");
    printMatrix(out, diff1);

    printf("\n");

    Matrix diff2 = diff(Matrix1, Matrix1);
    fprintf(out, "A-A = \n");
    printMatrix(out, diff2);

    printf("\n");

    Matrix trans = transpose(Matrix1);
    fprintf(out, "Transpose(A) = \n");
    printMatrix(out, trans);

    printf("\n");

    Matrix prod1 = product(Matrix1, Matrix2);
    fprintf(out, "A*B = \n");
    printMatrix(out, prod1);

    printf("\n");

    Matrix prod2 = product(Matrix2, Matrix2);
    fprintf(out, "B*B = \n");
    printMatrix(out, prod2);


    freeMatrix(&Matrix1);
    freeMatrix(&Matrix2);
	freeMatrix(&scalr);
	freeMatrix(&sum1);
	freeMatrix(&sum2);
	freeMatrix(&diff1);
	freeMatrix(&diff2);
	freeMatrix(&trans);
	freeMatrix(&prod1);
	freeMatrix(&prod2);


    return 0;
}

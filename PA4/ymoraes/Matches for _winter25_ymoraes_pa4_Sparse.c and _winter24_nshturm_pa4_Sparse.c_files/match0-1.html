#include <stdio.h>
#include <stdlib.h>

#include "Matrix.h"



int main(int argc, char *argv[]) {
        if (argc != 3) {
                printf("Usage: %s <input file> <output file>\n", argv[0]);
                return 1;
        }

        const char* fIn = argv[1];
        FILE* file = fopen(fIn, "r");
        if (file == NULL) {
                perror("error opening file");
                return 1;
        }

        const char* fOut = argv[2];
        FILE* output = fopen(fOut, "w");
        if (output == NULL) {
                perror("error opening file");
                return 1;
	}


        int n, a, b;

        fscanf(file, "%d %d %d\n", &n, &a, &b);

        Matrix mA = newMatrix(n);
        Matrix mB = newMatrix(n);

        int row, col;
        double value;

        for (int i = 0; i < a; i++) {
                fscanf(file, "%d %d %lf\n",&row, &col, &value);
                changeEntry(mA, row, col, value);
        }

        fscanf(file, "\n");

        for (int j = 0; j < b; j++) {
                fscanf(file, "%d %d %lf\n", &row, &col, &value);


                changeEntry(mB, row, col, value);
        }

        fclose(file);

        fprintf(output, "A has %d non-zero entries: \n", NNZ(mA));
        printMatrix(output, mA);

        printf("\n");

        fprintf(output, "B has %d non-zero entries: \n", NNZ(mB));
        printMatrix(output, mB);

        printf("\n");

        Matrix scal = scalarMult(1.5, mA);
        fprintf(output, "(1.5)*A =\n");
        printMatrix(output, scal);

        printf("\n");

        Matrix s1 = sum(mA, mB);
        fprintf(output, "A+B = \n");
        printMatrix(output, s1);

        printf("\n");

        Matrix s2 = sum(mA, mA);
        fprintf(output, "A+A = \n");
        printMatrix(output, s2);

        printf("\n");

        Matrix d1 = diff(mB, mA);
        fprintf(output, "B-A = \n");
        printMatrix(output, d1);

        printf("\n");

        Matrix d2 = diff(mA, mA);
        fprintf(output, "A-A = \n");
        printMatrix(output, d2);

        printf("\n");

        Matrix trnsp = transpose(mA);
        fprintf(output, "Transpose(A) = \n");
        printMatrix(output, trnsp);

        printf("\n");

        Matrix p1 = product(mA, mB);
        fprintf(output, "A*B = \n");
        printMatrix(output, p1);

        printf("\n");

        Matrix p2 = product(mB, mB);
        fprintf(output, "B*B = \n");
        printMatrix(output, p2);


        freeMatrix(&mA);
        freeMatrix(&mB);
	freeMatrix(&scal);
	freeMatrix(&s1);
	freeMatrix(&s2);
	freeMatrix(&d1);
	freeMatrix(&d2);
	freeMatrix(&trnsp);
	freeMatrix(&p1);
	freeMatrix(&p2);


        return 0;
}

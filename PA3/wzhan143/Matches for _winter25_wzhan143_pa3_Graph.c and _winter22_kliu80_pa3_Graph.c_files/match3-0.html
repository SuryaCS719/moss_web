#include <stdbool.h>
#include <stdio.h>
#include <stdint.h>
#include <stdlib.h>
#include "List.h"
#include "Graph.h"



typedef struct GraphObj{    
    int *color;
    int *parent;
    int *discover;
    int *finish;
    List *neighbors;
    int order;
    int size;
    int time;
} GraphObj;

bool isValidVertex(Graph G, int u) {
    return u >= 1 && u <= getOrder(G);
}

Graph newGraph(int n){
    Graph G = malloc(sizeof(GraphObj));
    if (G == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(1);
    }
    G->color = (int *)calloc(n+1, sizeof(int));
    G->parent = (int *)calloc(n+1, sizeof(int));


    G->discover = (int *)calloc(n+1, sizeof(int));
    G->finish = (int *)calloc(n+1, sizeof(int));
    G->neighbors = (List *)calloc(n+1, sizeof(List));
    if (G->color == NULL || G->parent == NULL || G->discover == NULL || G->finish == NULL || G->neighbors == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(1);
    }


    G->order = n;
    G->size = 0;
    G->time = 0;
    for(uint32_t i = 1; i < n+1; i++){
        G->neighbors[i] = newList();
        G->discover[i] = UNDEF;
        G->finish[i] = UNDEF;
    }
    return G;
}

void freeGraph(Graph *pG){
    if(pG != NULL && *pG != NULL){
        for(uint32_t i = 1; i <= (*pG)->order; i++){
            freeList(&(*pG)->neighbors[i]);
        }
        free((*pG)->color);
        free((*pG)->parent);
        free((*pG)->discover);
        free((*pG)->finish);
        free((*pG)->neighbors);
        free(*pG);
        *pG = NULL;
    }
}

int getOrder(Graph G){
    return G->order;
}

int getSize(Graph G){
    return G->size;
}

int getParent(Graph G, int u){
    if (isValidVertex(G, u)){
        return G->parent[u];
    }
    return NIL;
}

int getDiscover(Graph G, int u){
    if (isValidVertex(G, u)){
        return G->discover[u];
    }
    return NIL;
}

int getFinish(Graph G, int u){
    if (isValidVertex(G, u)){
        return G->finish[u];
    }
    return NIL;
}

void addEdge(Graph G, int u, int v){
    if (isValidVertex(G, u) && isValidVertex(G, v)){
        addArc(G, u, v);
        addArc(G, v, u);
    }
}

void addArc(Graph G, int u, int v){
    if (isValidVertex(G, u) && isValidVertex(G, v)){
        moveBack(G->neighbors[u]);


        while (index(G->neighbors[u]) >= 0 && get(G->neighbors[u]) > v){
            movePrev(G->neighbors[u]);
        }
        if (index(G->neighbors[u]) != -1){
            insertAfter(G->neighbors[u], v);
        }
        else{
            prepend(G->neighbors[u], v);
        }
        G->size++;
    }
}



void visit(Graph G, int vertex, List s){
    G->discover[vertex] = ++(G->time);
    G->color[vertex] = 1;
    moveFront(G->neighbors[vertex]);
    for (int i = 0; i < length(G->neighbors[vertex]); i++){
        int adjVertex = get(G->neighbors[vertex]);
        if (G->color[adjVertex] == 0){
            G->parent[adjVertex] = vertex;
            visit(G, adjVertex, s);
        }
        moveNext(G->neighbors[vertex]);
    }
    G->color[vertex] = 2;
    prepend(s, vertex);
    G->finish[vertex] = ++(G->time);
}

void DFS(Graph G, List s){
    if (length(s) == G->order){
        List originalList = copyList(s);
        moveFront(originalList);
        clear(s);
        for (int i = 1; i <= G->order; i++){


            G->color[get(originalList)] = 0;
            G->parent[get(originalList)] = NIL;
            moveNext(originalList);
        }
        G->time = 0;
        moveFront(originalList);
        for (int i = 1; i <= G->order; i++){


            if (G->color[get(originalList)] == 0){
                visit(G, get(originalList), s);
            }
            moveNext(originalList);
        }
        freeList(&originalList);
    }
}

Graph transpose(Graph G){
    Graph transposedGraph = newGraph(G->order);
    for (int i = 1; i <= G->order; i++){


        moveFront(G->neighbors[i]);
        for (int j = 0; j < length(G->neighbors[i]); j++){
            int v = get(G->neighbors[i]);
            addArc(transposedGraph, v, i);
            moveNext(G->neighbors[i]);
        }
    }
    return transposedGraph;
}

Graph copyGraph(Graph G){
    Graph copy = newGraph(G->order);
    for (int i = 1; i <= G->order; i++){
        copy->neighbors[i] = copyList(G->neighbors[i]);
        copy->discover[i] = G->discover[i];
        copy->color[i] = G->color[i];


        copy->parent[i] = G->parent[i];
    }
    copy->order = G->order;
    copy->size = G->size;
    copy->time = G->time;
    return copy;
}

void printGraph(FILE *out, Graph G){
    for (uint32_t i = 1; i <= G->order; i++){
        fprintf(out, "%u: ", i);
        printList(out, G->neighbors[i]);
        fprintf(out, "\n");
    }
}

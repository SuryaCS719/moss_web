#include "Graph.h"

#define INF -1
#define NIL -10
#define UNDEF -100

#define WHITE 0
#define GRAY 2
#define BLACK 1



int time = 0;

// Structs 


typedef struct GraphObj{
    List* list; 
    int* color;
    int* parent;
    int* distance;
    int* discover; 
    int* finish;
    int sourceVertex;
    int order;  
    int size; 
}GraphObj;

// Constructors-Destructors ---------------------------------------------------

// Returns a new point to the new graph(edges)
Graph newGraph(int n)
{
    Graph G = malloc(sizeof(GraphObj));
    assert(G != NULL);
    G->list = calloc(n, sizeof(List));
    G->color = calloc(n, sizeof(int));
    G->parent = calloc(n, sizeof(int)); 
    G->distance = calloc(n, sizeof(int));
    G->discover = calloc(n, sizeof(int)); 
    G->finish = calloc(n, sizeof(int)); 
    G->sourceVertex = NIL;
    G->order = n;
    G->size = 0;

    for(int i = 0; i < (n); i++)    
    {
        G->list[i] = newList();
        G->parent[i] = NIL;
        G->distance[i] = INF;
        G->discover[i] = UNDEF;
        G->finish[i] = UNDEF;

    }
    return G;
}

// Frees all heap memory associated with *pG, and sets *pG to NULL.
void freeGraph(Graph* pG)
{
    if(*pG == NULL) //check if G is NULL
    {
        printf("Graph Error: calling freeGraph() on NULL Graph reference\n");
        exit(EXIT_FAILURE);   
    }
    
    if(pG != NULL && *pG != NULL)
    {
        for(int i = 0; i < (*pG)->order; i++)
        {
            freeList(&((*pG)->list[i]));
        }
        free((*pG)->list);
        free((*pG)->distance);
        free((*pG)->parent);
        free((*pG)->color);
        free((*pG)->discover);
        free((*pG)->finish);
        free(*pG);
        *pG = NULL;
    }
}

int getParent(Graph G, int u)
{
    if(G == NULL)
    {
        printf("Graph Error: calling getParent() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }

    if(u < 1 || u > getOrder(G))
    {
        printf("Graph Error: calling getParent() on out of bounds vertex\n");
        exit(EXIT_FAILURE);
    }



    if(G->sourceVertex == NIL)
    {
        return NIL;
    }

    return G->parent[u-1];  //returns parent of vertex u
}

Graph transpose(Graph G)
{
    if(G == NULL)
    {
        printf("Graph Error: calling transpose() on NULL Graph reference\n");


        exit(EXIT_FAILURE);
    }

    Graph T = newGraph(getOrder(G));   
    List L = NULL;

    for(int i = 0; i < getOrder(G); i++)
    {
        L = G->list[i];
        for(moveFront(L); index(L) >= 0; moveNext(L))
        {
            addArc(T, get(L), i+1);
        }
    }

    return T;
}

Graph copyGraph(Graph G)
{
    if(G == NULL)
    {
        printf("Graph Error: calling copyGraph() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }
    
    Graph copy_graph = newGraph(getOrder(G)); //create new graph


    copy_graph->order = getOrder(G);
    copy_graph->size = getSize(G);
    copy_graph->sourceVertex = getSource(G);

    for(int i = 0; i < getOrder(G); i++)
    {
        copy_graph->list[i] = copyList(G->list[i]);    
        copy_graph->color[i] = G->color[i];
        copy_graph->parent[i] = G->parent[i];    
        copy_graph->distance[i] = G->distance[i];    


        copy_graph->discover[i] = G->discover[i];    
        copy_graph->finish[i] = G->finish[i];    
    }

    return copy_graph;
}

void getPath(List L, Graph G, int u)
{
    if(G == NULL || L == NULL)
    {
        printf("Graph Error: calling getPath() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }

    if(u < 1 || u > getOrder(G))
    {
        printf("Graph Error: calling getPath() on out of bounds vertex\n");
        exit(EXIT_FAILURE);
    }

    if(G->sourceVertex == NIL)
    {
        printf("Graph Error: calling getPath() on NIL source vertex\n");
        exit(EXIT_FAILURE);
    } 
    


    if(u != getSource(G) && getParent(G, u) == NIL)
    {
        append(L, NIL);
        return;
    }

    List temp = newList();

    int current = u;
    while(current != NIL)
    {
        prepend(temp, current);
        current = getParent(G, current);
    }

    int i = 0;
    moveFront(temp);
    while(i < length(temp))
    {
        append(L, get(temp));
        moveNext(temp);
        i++;
    }

    freeList(&temp);
}

// Manipulation procedures 

void makeNull(Graph G)
{
    if (G == NULL)
    {
        printf("Graph Error: calling makeNull() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }



    int i = 0;

    while(i < getOrder(G))
    {
        clear(G->list[i]); 
        G->color[i] = WHITE;
        G->parent[i] = NIL;
        G->distance[i] = INF;
        i++;
    }

    G->size = 0;
    G->sourceVertex = NIL;
}

void addEdge(Graph G, int u, int v) 
{
    if(G==NULL)
    {
        printf("Graph Error: calling addEdge() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }

    if (u < 1 || u > getOrder(G) || v < 1 || v > getOrder(G)) 
    {
        printf("Graph Error: calling addEdge() with out of bounds vertex\n");


        exit(EXIT_FAILURE);
    }

    //add edge to both u and v
    List U = G->list[u-1];
    List V = G->list[v-1];

    for(moveFront(U); index(U)>=0; moveNext(U))
        if(get(U) == v)
            return;

    if(length(U) == 0)
    {
        append(U, v);
    }
    else
    {
        if(v > back(U))
        {
            append(U, v);
        }
        else
        {
            moveFront(U);
            while(v > get(U))
            {
                moveNext(U);
            }
            insertBefore(U,v);
        }
    }

    if(length(V) == 0)
    {
        append(V, u);
    }
    else
    {
        if(u > back(V))
        {
            append(V, u);
        }
        else
        {
            moveFront(V);
            while(u > get(V))
            {                
                moveNext(V);
            }
            insertBefore(V,u);
        }
    }
    G->size++;
}

void addArc(Graph G, int u, int v)
{
    if(G==NULL)
    {
        printf("Graph Error: calling addArc() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }

    if(u < 1 || u > getOrder(G) || v < 1 || v > getOrder(G))
    {
        printf("Graph Error: calling addArc() with out of bounds vertex\n");
        exit(EXIT_FAILURE);
    }



    List U = G->list[u-1];

    //check if v is in U list
    for(moveFront(U); index(U)>=0; moveNext(U))
        if(get(U) == v)
            return;

    //main append loop
    if(length(U) == 0)
    {
        append(U, v);
    }
    else
    {
        if(v > back(U))
        {
            append(U, v);
        }
        else
        {
            moveFront(U);
            while(v > get(U))
                moveNext(U);
            insertBefore(U, v);
        }
    }
    G->size++;
}

void Visit(Graph G, List S, int x) {
    G->color[x] = GRAY;
    G->discover[x] = ++time; // Pre-increment to assign the current time

    List adjList = G->list[x];
    for (moveFront(adjList); index(adjList) >= 0; moveNext(adjList)) {
        int adjVertex = get(adjList) - 1; // Adjusting for 0-based index
        if (G->color[adjVertex] == WHITE) {
            G->parent[adjVertex] = x + 1; // Set parent before the recursive call
            Visit(G, S, adjVertex);
        }
    }



    G->color[x] = BLACK;
    G->finish[x] = ++time; // Post-increment for finish time
    prepend(S, x + 1); // Prepend to list for processing order
}

void DFS(Graph G, List S)
{
    if(G == NULL)
    {
        printf("Graph Error: calling DFS() on NULL Graph reference\n"); 
        exit(EXIT_FAILURE);
    }   

    if(length(S) != getOrder(G))
    {
        printf("Graph Error: calling DFS() with invalid List\n");
        exit(EXIT_FAILURE);
    }



    G->sourceVertex = 0; //Set to zero so getter functions dont return NIL

    for(int i = 0; i < getOrder(G); i++)
    {
        G->color[i] = WHITE;
        G->parent[i] = NIL;
    }
    time = 0;
    List L = copyList(S); //copy list S
    clear(S);

    for(moveFront(L); index(L) >= 0; moveNext(L))
    {
        if(G->color[get(L)-1] == WHITE)
            Visit(G, S, get(L)-1);
    }

    freeList(&L);
}

void BFS(Graph G, int s)
{
    if(G == NULL)
    {
        printf("Graph Error: calling BFS() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }



    G->sourceVertex = s;

    //sets all elements to undiscovered
    for(int i = 0; i < getOrder(G); i++)
    {
        G->color[i] = WHITE;
        G->distance[i] = INF;
        G->parent[i] = NIL;
    }

    //sets source vertex to discovered
    G->distance[s-1] = 0;
    G->color[s-1] = GRAY;
    G->parent[s-1] = NIL;

    int curr = 0;

    List L = newList();
    prepend(L, s);
    while(length(L) != 0)
    {
        moveFront(L);
        curr = get(L)-1;
        deleteFront(L);
        List temp = G->list[curr];
        moveFront(temp);
        while(index(temp) != -1)
        {
            int adj_index = get(temp)-1;
            if(G->color[adj_index] == WHITE)
            {
                G->color[adj_index] = GRAY;
                G->distance[adj_index] = G->distance[curr]+1;
                G->parent[adj_index] = curr+1;
                append(L, get(temp));
            }
            moveNext(temp);
        }
        G->color[curr] = BLACK;
    }
    freeList(&L);
}

void printGraph(FILE* out, Graph G)
{
    if(G == NULL)
    {
        printf("Graph Error: calling printGraph() on NULL Graph reference\n");
        exit(EXIT_FAILURE);
    }
    


    for(int i = 0; i < G->order; i++)
    {
        fprintf(out, "%d: ", i+1);
        printList(out, G->list[i]);
        fprintf(out, "\n");
    }
}

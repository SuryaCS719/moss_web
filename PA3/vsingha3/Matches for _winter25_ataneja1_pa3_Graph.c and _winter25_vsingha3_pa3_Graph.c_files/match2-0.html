#include <stdio.h>

#include "Graph.h"

typedef struct GraphObj {
    int order;
    int size;
    List* list;
    int* parent_arr;
    int* discover_arr;
    int* finish_arr;
} GraphObj;

void Visit(Graph G, int u, int* time, List S){
    G->discover_arr[u] = ++(*time);
    for (moveFront(G->list[u]); index(G->list[u]) >= 0; moveNext(G->list[u])) {
        int v = get(G->list[u]);


        if (G->discover_arr[v] == UNDEF) {
            G->parent_arr[v] = u;
            Visit(G, v, time, S);
        }
    }
    G->finish_arr[u] = ++(*time);
    prepend(S, u);
}

Graph newGraph(int n) {
    Graph G = malloc(sizeof(GraphObj));
    G->order = n;
    G->size = 0;
    G->list = calloc(n + 1, sizeof(List));
    G->parent_arr = calloc(n + 1, sizeof(int));
    G->discover_arr = calloc(n + 1, sizeof(int));
    G->finish_arr = calloc(n + 1, sizeof(int));
    for (int i = 1; i <= n; i++) {
        G->list[i] = newList();
        G->parent_arr[i] = NIL;
        G->discover_arr[i] = UNDEF;
        G->finish_arr[i] = UNDEF;
    }
    return G;
}

void freeGraph(Graph* pG) {
    if (pG != NULL && *pG != NULL) {
        for (int i = 1; i <= (*pG)->order; i++) {
            freeList(&((*pG)->list[i]));
        }
        free((*pG)->list);
        free((*pG)->parent_arr);
        free((*pG)->discover_arr);
        free((*pG)->finish_arr);
        free(*pG);
        *pG = NULL;
    }
}

int getParent(Graph G, int u) {
    if (G == NULL || u < 1 || u > G->order) {
        fprintf(stderr, "Error: Calling getParent() with invalid vertex\n");
        exit(EXIT_FAILURE);
    }
    return G->parent_arr[u] == 0 ? NIL : G->parent_arr[u];
}

        fprintf(stderr, "Error: Calling getDiscover() with invalid vertex\n");
        exit(EXIT_FAILURE);
    }
    return G->discover_arr[u];
}

    if (G == NULL || u < 1 || u > G->order) {
        fprintf(stderr, "Error: Calling getFinish() with invalid vertex\n");
        exit(EXIT_FAILURE);
    }
    return G->finish_arr[u];
}


void addEdge(Graph G, int u, int v) {


    if (u < 1 || u > G->order || v < 1 || v > G->order) {
        fprintf(stderr, "Graph Error: calling addEdge() with invalid vertices %d, %d\n", u, v);
        exit(EXIT_FAILURE);
    }
    addArc(G, u, v);
    addArc(G, v, u);
    G->size--;
}

void addArc(Graph G, int u, int v) {
    if (u < 1 || u > G->order || v < 1 || v > G->order) {
        fprintf(stderr, "Graph Error: calling addArc() with invalid vertices %d, %d\n", u, v);
        exit(EXIT_FAILURE);
    }
    List listU = G->list[u];
    moveFront(listU);
    while (index(listU) >= 0 && get(listU) < v) {
        moveNext(listU);
    }
    if (index(listU) == -1) {
        append(listU, v);
    } else {
        insertBefore(listU, v);
    }
    G->size++;
}
void DFS(Graph G, List S) {

    if (length(S) != G->order) {
        fprintf(stderr, "DFS Error: List size does not match graph order.\n");
        exit(EXIT_FAILURE);
    }
    
    for (int i = 1; i <= G->order; i++) {
        G->parent_arr[i] = NIL;
        G->discover_arr[i] = UNDEF;
        G->finish_arr[i] = UNDEF;
    }
    
    int time = 0;
    List temp = copyList(S);
    clear(S);
    
    for (moveFront(temp); index(temp) >= 0; moveNext(temp)) {


        int u = get(temp);
        if (G->discover_arr[u] == UNDEF) {
            Visit(G, u, &time, S);
        }
    }
    freeList(&temp);    
}

Graph transpose(Graph G) {
    Graph T = newGraph(G->order);
    for (int u = 1; u <= G->order; u++) {
        for (moveFront(G->list[u]); index(G->list[u]) >= 0; moveNext(G->list[u])) {


            int v = get(G->list[u]);
            addArc(T, v, u);
        }
    }
    return T;
}
Graph copyGraph(Graph G) {
    Graph C = newGraph(G->order);
    for (int u = 1; u <= G->order; u++) {
        for (moveFront(G->list[u]); index(G->list[u]) >= 0; moveNext(G->list[u])) {


            int v = get(G->list[u]);
            addArc(C, u, v);
        }
    }
    return C;
}
void printGraph(FILE* out, Graph G) {
    for (int i = 1; i <= G->order; i++) {
        fprintf(out, "%d: ", i);
        printList(out, G->list[i]);
        fprintf(out, "\n");
    }

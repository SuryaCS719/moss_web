#include <stdio.h>
#include <stdlib.h>

#include "Graph.h"


// Structs -----


typedef struct GraphObj {
    List* adj;
    int order;
    int size;
    int* color;
    int* parent;
    int* d;
    int* finish;
} GraphObj;


// Constructors and Destructors -----
Graph newGraph(int n) {
    Graph G = malloc(sizeof(GraphObj));
    G->adj = malloc((n + 1) * sizeof(List));
    G->order = n;
    G->size = 0;
    G->color = malloc((n + 1) * sizeof(int));
    G->parent = malloc((n + 1) * sizeof(int));
    G->d = malloc((n + 1) * sizeof(int));
    G->finish = malloc((n + 1) * sizeof(int));

    for (int i = 0; i < n+1; i++) {
        G->adj[i] = newList();
        G->parent[i] = NIL;
        G->d[i] = INF;
        G->finish[i] = UNDEF;
    } return G;
}

void freeGraph(Graph* pG) {
    if (pG != NULL && *pG != NULL) {
        Graph G = *pG;
        for (int i = 0; i <= G->order; i++){
            freeList(&(G->adj[i]));
        }
        free(G->adj);
        free(G->color);
        free(G->parent);
        free(G->d);
        free(G->finish);
        free(G);
        *pG = NULL;
    }
}


// Access Functions -----
int getOrder(Graph G) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling getOrder() on a NULL Graph.");
        exit(EXIT_FAILURE);
    }
    return G->order;
}

int getSize(Graph G) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling getSize() on a NULL Graph.");
        exit(EXIT_FAILURE);
    }
    return G->size;
}

int getParent(Graph G, int u) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling getParent() on a NULL Graph.");
        return NIL;
    }
    return G->parent[u];
}

int getDiscover(Graph G, int u) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling getDiscover() on a NULL Graph.");
        return INF;
    }
    return G->d[u];
}

int getFinish(Graph G, int u) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling getFinish() on a NULL Graph.");


        return NIL;
    }
    if (!(1 <= u && u <= getOrder(G))) {
        fprintf(stderr, "Error: getFinish() violated precondition: u is not in range [1, getOrder(G)].");
        exit(EXIT_FAILURE);
    }
    return G->finish[u];
}

// Manipulation Procedures -----
void addArc(Graph G, int u, int v) {
    if (G == NULL) {
        fprintf(stderr, "Error: addArc() is Calling on a NULL Graph.");
        exit(EXIT_FAILURE);
    }
    if (1 > u || u > getOrder(G)) {
        return;
        exit(EXIT_FAILURE);
    }
    if (1 > v || v > getOrder(G)) {
        return;
        exit(EXIT_FAILURE);
    }

    moveFront(G->adj[u]);
    for (int i = 1; i < length(G->adj[u]); i++){


        if (get(G->adj[u]) == v) {
            return;
        } moveNext(G->adj[u]);
    }
    moveFront(G->adj[u]);
    
    while (index(G->adj[u]) != NIL && get(G->adj[u]) < v){
        moveNext(G->adj[u]);
    }
    if (index(G->adj[u]) == NIL) {
        append(G->adj[u], v);
    }
    else{
        insertBefore(G->adj[u], v);
    } G->size++;
}

void addEdge(Graph G, int u, int v) {
    addArc(G, u, v);
    addArc(G, v, u);
    G->size--;
}

void Visit(Graph G, List S, int u, int time) {
    time++;
    G->d[u] = time;
    G->color[u] = 1;
    int t;


    for (moveFront(G->adj[u]); index(G->adj[u]) >= 0; moveNext(G->adj[u])) {
        t = get(G->adj[u]);
        if (G->color[t] == 0) {
            G->parent[t] = u;
            Visit(G, S, t, time);
        }
    }
    G->color[u] = 2;
    time++;
    G->finish[u] = time;
    
    prepend(S, u);
}

void DFS(Graph G, List S) {


    if (G == NULL || S= NULL) {
        fprintf(stderr, "Error: Calling DFS() on a NULL Graph.");
        exit(EXIT_FAILURE);
    } 
    if (length(S) != getOrder(G)) {
        fprintf(stderr, "Error: Length(S) must be equal to getOrder(G)");
        exit(EXIT_FAILURE);
    }
    List copyStack = copyList(S);
    clear(S);
    int time = 0;
    for (moveFront(copyStack); index(copyStack) >= 0; moveNext(copyStack)) {
        int u = get(copyStack);
        G->color[u] = WHITE;
        G->parent[u] = NIL;
        G->d[u] = UNDEF;
        G->finish[u] = UNDEF;
    }
    for (moveFront(copyStack); index(copyStack) >= 0; moveNext(copyStack)) {
        int v = get(copyStack);
        if (G->color[v] == WHITE) {
            Visit(G, S, v, &time);
        }
    }
    freeList(&copyStack);
}


// Other Functions -----
Graph transpose(Graph G) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling transpose() on a NULL Graph.");
        exit(EXIT_FAILURE);
    }
    Graph H = newGraph(getOrder(G));
    for (int i = 1; i <= getOrder(G); i++) {
        List L = (G->adj[i]);
        moveFront(L);
        while (index(L) >= 0 && get(L)!=-1) {
            addArc(H, get(L), i);
            moveNext(L);
        }
    } H->size = G->size;
    return H;
}

Graph copyGraph(Graph G) {


    if (G == NULL) {
        fprintf(stderr, "Error: Calling copyGraph() on a NULL Graph.");
        return NULL;
    }
    Graph copy = newGraph(G->order);
    for (int i = 1; i <= getOrder(G); i++) {
        moveFront(G->adj[i]);
        while(index(G->adj[i]) != NIL) {
            addArc(copy, i, get(G->adj[i]));
            moveNext(G->adj[i]);
        }
    }
    copy->size = G->size;
    return copy;
}

void printGraph(FILE* out, Graph G) {
    if (G == NULL) {
        fprintf(stderr, "Error: Calling printGraph() on a NULL Graph.");
        exit(EXIT_FAILURE);
    }    

    for (int i = 1; i<= getOrder(G); i++) {
        fprintf(out, "%d:", i);
        moveFront(G->adj[i]);
        
        while (index(G->adj[i]) != NIL) {
            fprintf(out, " %d", get(G->adj[i]));
            moveNext(G->adj[i]);
        } fprintf(out, "\n");
    }
}

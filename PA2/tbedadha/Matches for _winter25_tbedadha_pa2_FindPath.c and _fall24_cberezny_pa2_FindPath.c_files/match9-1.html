#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<string.h>

#include"Graph.h"



int main(int argc, char * argv[]){
    FILE *in, *out;
    if( argc != 3 ){
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }

    // open files for reading and writing 
    in = fopen(argv[1], "r");
    if( in==NULL ){
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }

    out = fopen(argv[2], "w");
    if( out==NULL ){
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }

    // Read vertices/edges
    int n;
    fscanf(in, "%d", &n);
    Graph G = newGraph(n);

    int vertex1;


    int vertex2;
    fscanf(in, "%d %d\n", &vertex1, &vertex2);
    while(vertex1 != 0 || vertex2 != 0){
        addEdge(G, vertex1, vertex2);
        fscanf(in, "%d %d\n", &vertex1, &vertex2);
    }

    maintainGraph(G);
    printGraph(out, G);

    // Read start/end pairs
    int start;
    int end;
    fscanf(in, "%d %d\n", &start, &end);

    while(start != 0 || end != 0){
        if(start == end){
            fprintf(out, "\nThe distance from %d to %d is 0\n", start, end);
            fprintf(out, "A shortest %d-%d path is: %d\n", start, end, start);
            fscanf(in, "%d %d\n", &start, &end);
            continue;
        }
        BFS(G, start);
        // Print distance
        int dist = getDist(G, end);
        if(dist == INF){
            fprintf(out, "\nThe distance from %d to %d is infinity\n", start, end);
        }else{
            fprintf(out, "\nThe distance from %d to %d is %d\n", start, end, dist);
        }
        // Find shortest path from parent
        List path = newList();
        getPath(path, G, end);
        if(front(path) == NIL){
            fprintf(out, "No %d-%d path exists\n", start, end);
        }else{
            fprintf(out, "A shortest %d-%d path is: ", start, end);
            printList(out, path);
        }
        freeList(&path);
        fscanf(in, "%d %d\n", &start, &end);
    }

    fclose(in);
    fclose(out);
    
    freeGraph(&G);
    return(0);
}

#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include<string.h>

#include"Graph.h"



int main(int argc, char * argv[]){
    FILE *in, *out;
    if( argc != 3 ){
        fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
        exit(1);
    }
 
    in = fopen(argv[1], "r");
    if( in==NULL ){
        fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
        exit(1);
    }

    out = fopen(argv[2], "w");
    if( out==NULL ){
        fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
        exit(1);
    }

    int n;
    fscanf(in, "%d", &n);
    Graph G = newGraph(n);

    int v1,v2;
    fscanf(in, "%d %d\n", &v1, &v2);
    while(v1 != 0 || v2 != 0){
        addEdge(G, v1, v2);
        fscanf(in, "%d %d\n", &v1, &v2);
    }

    sortGraph(G);
    printGraph(out, G);

    int source,dest;
    fscanf(in, "%d %d\n", &source, &dest);

    while(source != 0 || dest != 0){
        if(source == dest){
            fprintf(out, "\nThe distance from %d to %d is 0\n", source, dest);
            fprintf(out, "A shortest %d-%d path is: %d\n", source, dest, source);
            fscanf(in, "%d %d\n", &source, &dest);
            continue;
        }

        BFS(G, source);
        int dist = getDist(G, dest);
        if(dist == INF){
            fprintf(out, "\nThe distance from %d to %d is infinity\n", source, dest);
        }else{
            fprintf(out, "\nThe distance from %d to %d is %d\n", source, dest, dist);
        }

        List path = newList();
        getPath(path, G, dest);
        if(front(path) == NIL){
            fprintf(out, "No %d-%d path exists\n", source, dest);
        }else{
            fprintf(out, "A shortest %d-%d path is: ", source, dest);
            printList(out, path);
        }
        freeList(&path);
        fscanf(in, "%d %d\n", &source, &dest);
    }

    fclose(in);
    fclose(out);
    freeGraph(&G);
    return(0);
}

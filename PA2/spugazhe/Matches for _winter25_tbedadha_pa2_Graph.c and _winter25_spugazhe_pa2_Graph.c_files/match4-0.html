#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>

#include"Graph.h"



typedef struct GraphObj{
	List* neighbors;
	List color;
	List parent;
	List distance;
	int order;
	int size;
	int source;
} GraphObj;

/*** Constructors-Destructors ***/
Graph newGraph(int n){
	Graph g = (GraphObj *)malloc(1 * sizeof(GraphObj));
	g->neighbors = (List*)calloc(n + 1, sizeof(List));
	
	for(int i = 0; i <= n; i++){
		g->neighbors[i] = newList();
	}

	g->color = newList();
	g->parent = newList();
	g->distance = newList();
	
	for(int i = 0; i <= n; i++){
		append(g->color, WHITE);
		append(g->parent, NIL);
		append(g->distance, INF);
	}

	g->order = n;
	g->size = 0;
	g->source = NIL;
	return g;
}

void freeGraph(Graph* pG){
	if(pG != NULL && *pG != NULL){
		
		for(int i = 0; i <= (*pG)->order; i++){
			freeList(&((*pG)->neighbors[i]));
		}

		freeList(&((*pG)->color));
		freeList(&((*pG)->parent));
		freeList(&((*pG)->distance));
		free((*pG)->neighbors);
		free(*pG);
		*pG = NULL;
	}
}

bool called = false;

/*** Access Functions ***/
int getOrder(Graph G){
	return G->order;
}

int getSize(Graph G){
	return G->size;
}

int getSource(Graph G){
	return G->source;
}

int getParent(Graph G, int u){
	


	if(called == false) {
		return NIL;
	}

	List parent = G->parent;
	moveFront(parent);
	
	while( index(parent) != u ){
		moveNext(parent);
	}

	return get(parent);
}

int getDist(Graph G, int u){
	


	if(called == false) {
		return NIL;
	}

	List dist = G->distance;
	moveFront(dist);
	
	while( index(dist) != u ){
		moveNext(dist);
	}

	return get(dist);
}

void getPath(List L, Graph G, int u){
    
    if(u == getSource(G)){
        append(L, u);
        return;
    }

    List newL = newList();
    append(newL, u);
    int p = getParent(G,u);
    append(newL, p);
    
    while(p != getSource(G) && p != NIL){
        p = getParent(G, p);
        append(newL, p);
    }

    if(p == NIL){
        append(L, NIL);
    } else {
        moveBack(newL);
        
	while (index(newL) >= 0)
        {
            append(L,get(newL));
            movePrev(newL);
        }

    }
    freeList(&newL);
}

/*** Manipulation Procedures ***/
void makeNull(Graph G){
	clear(G->color);
	clear(G->distance);
	clear(G->parent);
	
	for(int i = 0; i <= G->order; i++){
		clear(G->neighbors[i]);
	}

	G->order = 0;
	G->source = NIL;
}

void addEdge(Graph G, int u, int v){
	append(G->neighbors[u], v);
	append(G->neighbors[v], u);
	G->size += 1;
}

void addArc(Graph G, int u, int v){
	append(G->neighbors[u], v);
	G->size += 1;
}



void sortGraph(Graph G){
    for(int i = 1; i <= getOrder(G); i++){
        if(length(G->neighbors[i]) == 0){
            continue;
        }        
        List L = newList();
        moveFront(G->neighbors[i]);
        append(L, get(G->neighbors[i]));
        moveNext(G->neighbors[i]);
        while(index(G->neighbors[i]) >= 0){
            moveFront(L);
            while(index(L) >= 0){
                if(get(L) > get(G->neighbors[i])){
                    break;
                }
                moveNext(L);
            }
            if(index(L) < 0){
                append(L, get(G->neighbors[i]));
            }else{
                insertBefore(L, get(G->neighbors[i]));
            }
            moveNext(G->neighbors[i]);
        }
        freeList(&(G->neighbors[i]));
        G->neighbors[i] = L;
    }
}

void BFS(Graph G, int s){
    sortGraph(G);
    called = true;


    G->source = s;
    moveFront(G->color);
    moveFront(G->distance);
    moveFront(G->parent);

    for(int i = 0; i <= G->order; i++){
        set(G->color, WHITE);
        set(G->distance, INF);
        set(G->parent, NIL);
        moveNext(G->color);
        moveNext(G->distance);
        moveNext(G->parent);
    }

    moveFront(G->color);
    moveFront(G->distance);
    moveFront(G->parent);
    
    while (index(G->color) != s)
    {
        moveNext(G->color);
        moveNext(G->distance);
        moveNext(G->parent);
    }

    set(G->color, GRAY);
    set(G->distance, 0);
    set(G->parent, NIL);
    List queue = newList();
    append(queue, s);
    
    while(length(queue) > 0){
        int first = front(queue);
        deleteFront(queue);
        moveFront(G->neighbors[first]);
       
        while (index(G->neighbors[first]) >= 0){
            int adj = get(G->neighbors[first]);
            moveFront(G->distance);
            while(index(G->distance) != first){
                moveNext(G->distance);
            }
            int x_dist = get(G->distance);
       
            moveFront(G->color);
            moveFront(G->distance);
            moveFront(G->parent);
            
	    while(index(G->color) != adj){
                moveNext(G->color);
                moveNext(G->distance);
                moveNext(G->parent);
            }

            if(get(G->color) == WHITE){
                set(G->color, GRAY);
                set(G->distance, x_dist + 1);
                set(G->parent, first);
                append(queue, adj);
            }
            moveNext(G->neighbors[first]);
        }
        moveFront(G->color);
        
	while (index(G->color) != first)
        {
            moveNext(G->color);
        }
        set(G->color, BLACK);
    }
    freeList(&queue);
}

void printGraph(FILE* out, Graph G){


    for(int i = 1; i <= getOrder(G); i++){
        moveFront(G->neighbors[i]);
        fprintf(out, "%d:", i);
    
    	while(index(G->neighbors[i]) >= 0){
            int neighbor = get(G->neighbors[i]);
            fprintf(out, " %d", neighbor);
            moveNext(G->neighbors[i]);
        }
        fprintf(out, "\n");
    }
}

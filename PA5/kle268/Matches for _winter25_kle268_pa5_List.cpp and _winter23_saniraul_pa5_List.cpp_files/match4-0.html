// Class Constructors & Destructors ----------------------------------------

// Node Constructor
List::Node::Node(ListElement x) {
    data = x;
    prev = nullptr;
    next = nullptr;
}

// Default List Constructor
List::List() {
    frontDummy = new Node(INT_MIN);
    backDummy = new Node(INT_MAX);

    frontDummy->next = backDummy;
    backDummy->prev = frontDummy;

    beforeCursor = frontDummy;
    afterCursor = backDummy;

    pos_cursor = 0;
    num_elements = 0;
}

// Copy List Constructor
List::List(const List& L) {
    frontDummy = new Node(INT_MIN);
    backDummy = new Node(INT_MAX);

    frontDummy->next = backDummy;
    backDummy->prev = frontDummy;

    beforeCursor = frontDummy;
    afterCursor = backDummy;

    pos_cursor = 0;


    num_elements = 0;

    Node* N = L.frontDummy->next;
    while (N != L.backDummy) {
        int x = N->data;
        this->insertBefore(x);
        N = N->next;
    }

    this->moveFront();
    while (this->pos_cursor != L.pos_cursor) {
        this->moveNext();
    }
}

// List Destructor
List::~List() {
    this->clear();

    delete this->frontDummy;
    delete this->backDummy;
}

// Access functions --------------------------------------------------------

// Function length()
// Returns the length of this List.
int List::length() const {
    return this->num_elements;
}

// Function front()
// Precondition: length() > 0
// Returns the front element in this List.
ListElement List::front() const {
    if (!(this->length() > 0)) {
        cout << "List Error: calling front() when precondition is not met" << endl;
        return EXIT_FAILURE;
    }

    return frontDummy->next->data;
}

// Function back()
// Precondition: length() > 0
// Returns the back element in this List.
ListElement List::back() const {
    if (!(this->length() > 0)) {
        cout << "List Error: calling back() when precondition is not met" << endl;
        return EXIT_FAILURE;
    }

    return backDummy->prev->data;
}

// Function position()
// Returns the position of cursor in this List: 0 <= position() <= length().
int List::position() const {
    return this->pos_cursor;
}

// Function peekNext()
// Precondition: position() < length()
// Returns the element after the cursor.
ListElement List::peekNext() const {
    if (!(position() < length())) {
        cout << "List Error: calling peekNext() when precondition is not met" << endl;
        return EXIT_FAILURE;
    }

    return afterCursor->data;
}

// Function peekPrev()
// Precondition: position()>0
// Returns the element before the cursor.
ListElement List::peekPrev() const {
    if (!(position() > 0)) {
        cout << "List Error: calling peekPrev() when precondition is not me" << endl;


        return EXIT_FAILURE;
    }

    return beforeCursor->data;
}

// Manipulation procedures -------------------------------------------------

// Function clear()
// Deletes all elements in this List, setting it to the empty state.
void List::clear() {
    this->moveFront();

    while (this->length() > 0) {
        this->eraseAfter();
    }
}

// Function moveFront()
// Moves cursor to position 0 in this List.
void List::moveFront() {
    beforeCursor = frontDummy;
    afterCursor = beforeCursor->next;

    this->pos_cursor = 0;
}

// Function moveBack()
// Moves cursor to position length() in this List.
void List::moveBack() {
    afterCursor = backDummy;
    beforeCursor = afterCursor->prev;

    this->pos_cursor = this->length();
}

// Function moveNext()
// Precondition: position() < length()
// Advances cursor to next higher position. Returns the List element that was passed over. 
ListElement List::moveNext() {
    if (!(this->position() < this->length())) {
        cout << "List Error: calling moveNext() when precondition is not met" << endl;
        return EXIT_FAILURE;
    }

    beforeCursor = beforeCursor->next;
    afterCursor = afterCursor->next;

    this->pos_cursor++;

    return beforeCursor->data;
}

// Function movePrev()
// Precondition: position()>0
// Advances cursor to next lower position. Returns the List element that was passed over.
ListElement List::movePrev() {
    if (!(this->position() > 0)) {
        cout << "List Error: calling movePrev() when precondition is not met" << endl;


        return EXIT_FAILURE;
    }

    beforeCursor = beforeCursor->prev;
    afterCursor = afterCursor->prev;

    this->pos_cursor--;

    return afterCursor->data;
}

// Function insertAfter()
// Inserts x after cursor.
void List::insertAfter(ListElement x) {
    Node* N = new Node(x);

    N->prev = beforeCursor;
    N->next = afterCursor;

    beforeCursor->next = N;
    afterCursor->prev = N;

    afterCursor = N;

    this->num_elements++;
}

// Function insertBefore()
// Inserts x before cursor.
void List::insertBefore(ListElement x) {
    Node* N = new Node(x);

    N->prev = beforeCursor;
    N->next = afterCursor;

    beforeCursor->next = N;
    afterCursor->prev = N;

    beforeCursor = N;

    this->pos_cursor++;
    this->num_elements++;
}

// Function setAfter()
// Precondition: position() > 0
// Overwrites the List element after the cursor with x.
void List::setAfter(ListElement x) {
    if (!(this->position() > 0)) {
        throw range_error("List Error: calling setAfter() when precondition is not met");
    }

    afterCursor->next->data = x;
}

// Function setBefore()
// Precondition: position() > 0
// Overwrites the List element before the cursor with x.
void List::setBefore(ListElement x) {
    if (!(this->position() > 0)) {
        throw range_error("List Error: calling setBefore() when precondition is not met");
    }
    
    beforeCursor->prev->data = x;
}

// Function eraseAfter()
// Precondition: position()<length()
// Deletes element after cursor.
void List::eraseAfter() {
    if (!(this->position() < this->length())) {
        throw range_error("List Error: calling eraseAfter() when precondition is not met");
    }

    Node* N = afterCursor;

    beforeCursor->next = afterCursor->next;
    N->next->prev = beforeCursor;
    afterCursor = N->next;

    //afterCursor = afterCursor->next;
    //beforeCursor->next = afterCursor;
    //afterCursor->prev = beforeCursor;

    this->num_elements--;

    delete N;
}

// Function eraseBefore()
// Precondition: position() > 0
// Deletes element before cursor.
void List::eraseBefore() {
    if (!(this->position() > 0)) {
        throw range_error("List Error: calling eraseBefore() when precondition is not met");
    }



    Node* N = beforeCursor;

    beforeCursor = N->prev;
    beforeCursor->next = afterCursor;
    afterCursor->prev = beforeCursor;

    //beforeCursor = beforeCursor->prev;
    //beforeCursor->next = afterCursor;
    //afterCursor->prev = beforeCursor;

    this->num_elements--;
    this->pos_cursor--;

    delete N;
}

// Function findNext()
// Starting from the current cursor position, performs a linear search (in 
// the direction front-to-back) for the first occurrence of element x. If x
// is found, places the cursor immediately after the found element, then 
// returns the final cursor position. If x is not found, places the cursor 
// at position length(), and returns -1. 
int List::findNext(ListElement x) {
    Node* N = this->afterCursor;

    while (N != this->backDummy) {
        if (N->data == x) {
            this->beforeCursor = N;
            this->afterCursor = N->next;
            this->pos_cursor++;

            return this->pos_cursor;
        }

        N = N->next;
        this->pos_cursor++;
    }

    this->afterCursor = backDummy;
    this->beforeCursor = afterCursor->prev;
    this->pos_cursor = this->length();

    return -1;
}

// Function findPrev()
// Starting from the current cursor position, performs a linear search (in 
// the direction back-to-front) for the first occurrence of element x. If x
// is found, places the cursor immediately before the found element, then
// returns the final cursor position. If x is not found, places the cursor 
// at position 0, and returns -1. 
int List::findPrev(ListElement x) {
    Node* N = this->beforeCursor;

    while (N != this->frontDummy) {
        if (N->data == x) {
            this->beforeCursor = N->prev;
            this->afterCursor = N;
            this->pos_cursor--;

            return this->pos_cursor;
        }

        N = N->prev;
        this->pos_cursor--;
    }



    this->beforeCursor = frontDummy;
    this->afterCursor = beforeCursor->next;
    this->pos_cursor = 0;

    return -1;
}

// Function cleanup()
// Removes any repeated elements in this List, leaving only unique elements.
// The order of the remaining elements is obtained by retaining the frontmost 
// occurrance of each element, and removing all other occurances. The cursor 
// is not moved with respect to the retained elements, i.e. it lies between 
// the same two retained elements that it did before cleanup() was called.
void List::cleanup() {
    Node* N = this->frontDummy->next;
    int outter_pos = 0;

    while (N != this->backDummy) {
        int x = N->data;
        Node* temp = N->next;
        int inner_pos = outter_pos + 1;

        while (temp != this->backDummy) {
            int y = temp->data;
            Node* before = temp->prev;
            Node* after = temp->next;
            Node* del = temp;
            temp = after;

            if (x == y) {
                before->next = after;
                after->prev = before;

                delete del;
                this->num_elements--;

                if (inner_pos <= pos_cursor) {
                    pos_cursor--;
                }
            }

            inner_pos++;
        }

        outter_pos++;
        N = N->next;
    }

    int n = this->pos_cursor;
    this->moveFront();

    while (this->position() != n) {
        this->moveNext();
    }
}

// Function concat()
// Returns a new List consisting of the elements of this List, followed by
// the elements of L. The cursor in the returned List will be at postion 0.
List List::concat(const List& L) const {
    List newL;
    newL.moveBack();

    Node* N = this->frontDummy->next;

    while (N != this->backDummy) {
        newL.insertBefore(N->data);

        N = N->next;
    }

    N = L.frontDummy->next;

    while (N != L.backDummy) {
        newL.insertBefore(N->data);

        N = N->next;
    }

    newL.beforeCursor = newL.frontDummy;
    newL.afterCursor = newL.beforeCursor->next;
    newL.pos_cursor = 0;

    return newL;
}

// Function to_string()
// Returns a string representation of this List consisting of a comma 
// separated sequence of elements, surrounded by parentheses.
std::string List::to_string() const {
    Node* N;
    string s = "";

    s += "(";
    for (N = frontDummy->next; N != this->backDummy; N = N->next) {
        if (N->data != INT_MAX && N->data != INT_MIN) {
            s += std::to_string(N->data);
        }

        if (N->next != this->backDummy){
            s += ", " ;
        }
    }
    s += ")";

    return s;
}

// Function equals()
// Returns true if and only if this List is the same integer sequence as R.
// The cursors in this List and in R are unchanged.
bool List::equals(const List& R) const {
    if (this->num_elements != R.num_elements) {
        return false;
    }

    Node* N = this->frontDummy;
    Node* M = R.frontDummy;

    while (N != this->backDummy && M != R.backDummy) {
        if (N->data != M->data) {
            return false;
        }

        N = N->next;
        M = M->next;
    }

    return true;
}

// Overriden Operators -----------------------------------------------------

// Operator<< overloading
std::ostream& operator<<(std::ostream& stream, const List& L) {
    return stream << L.to_string();
}

// Operator== overloading
bool operator==(const List& A, const List& B) {
    return A.equals(B);
}

// Operator= overloading
List& List::operator=(const List& L) {
    if (this != &L) {
        List newL = L;

        swap(frontDummy, newL.frontDummy);
        swap(backDummy, newL.backDummy);

        swap(beforeCursor, newL.beforeCursor);
        swap(afterCursor, newL.afterCursor);

        swap(pos_cursor, newL.pos_cursor);
        swap(num_elements, newL.num_elements);
    }

    return *this;
}

#include<iostream>
#include<string>
#include<stdexcept>
#include<iomanip>
#include"List.h"



using namespace std;

// Function prototypes --------------------------------------------------------

void shuffle(List &D);

// maino()
int main(int argc, char *argv[]) {
	if (argc != 2 || stoi(argv[1]) < 1) {
		printf("Usage: ./Shuffle <integer>\n");
		exit(1);
	}

	List D; // original deck
	List S; // shuffled deck
	int n = stoi(argv[1]); // deck size
	
	// header for output
	cout << "deck size       shuffle count\n";


	cout << "------------------------------\n";

	for (int i = 1; i <= n; i++) {
		// count for the number of shuffles
		int s = 1;

		D.insertBefore(i); // fill the deck

		// copy D into S
		S = D;

		// begin shuffling
		shuffle(S);
		
		// keep shuffling until back in original order
		while(!(D == S)) {
			shuffle(S);
			s++;
		}

		cout << " " << std::left << std::setw(16) << i << s << endl;
	}

	return 0;
}

// shuffle()
// Shuffle the given list by splitting it in half,
// then "merging" it by alternately adding one value
// from each half, starting with the right half
void shuffle(List &D) {
	if (D.length() > 1) {
		// midpoint of D
		int mid = D.length() / 2;

		// remainder if deck size is not even
		int remain = D.length() % 2;

		// Create Lists to simulate left and right halves of D
		List L, R;
		D.moveFront();

		// Fill both halves, starting with the left half
		for (int i = 0; i < mid; i++) {
			L.insertBefore(D.moveNext());
		}
		for (int i = mid; i < D.length(); i++) {
			R.insertBefore(D.moveNext());
		}

		// Empty D
		D.clear();

		// Simulate the perfect shuffle by alternating adding values
		// from L and R, starting with R
		L.moveFront();
		R.moveFront();

		for (int i = 0; i < mid; i++) {
			D.insertBefore(R.moveNext());
			D.insertBefore(L.moveNext());
		}

		if (((mid * 2) + remain) % 2 != 0) {
			D.insertBefore(R.moveNext());
		}
	}
}

#include<iostream>
#include<string>
#include"List.h"

using namespace std;

#define DUMMY 0

// Node constructor: Initializes a new Node with given data, and null pointers for next and previous nodes.
List::Node::Node(int x){
   data = x;
   prev = nullptr;
   next = nullptr;
}

// List default constructor: Initializes an empty list with dummy nodes at the front and back, and sets the cursor to the start.
List::List(){
    frontDummy = new Node(DUMMY);
    backDummy = new Node(DUMMY);
    frontDummy->next = backDummy;
    backDummy->prev = frontDummy;
    beforeCursor = frontDummy;
    afterCursor = backDummy;
    pos_cursor = 0;
    num_elements = 0;
}

// Copy constructor: Creates a deep copy of a given list L.
List::List(const List& L){ 
    // Initial setup similar to the default constructor.
    frontDummy = new Node(DUMMY);
    backDummy = new Node(DUMMY);


    frontDummy->next = backDummy;
    backDummy->prev = frontDummy;
    beforeCursor = frontDummy;
    afterCursor = backDummy;
    num_elements = 0;

    // Copy the elements from the given list L.
    Node* N = L.frontDummy->next;
    while(N != L.backDummy){
        int data = N->data;
        this->insertBefore(data);
        N = N->next;
    }
    // Adjust the cursor position to match that of L.
    this->moveFront();
    while(this->pos_cursor != L.pos_cursor){
        this->moveNext();
    }
}

// Destructor: Clears the list and deletes the dummy nodes.
List::~List(){
    this->clear();
    delete this->frontDummy;
    delete this->backDummy;
}

// Returns the number of elements in the list.
int List::length() const{
    return this->num_elements;
}

// Returns the data of the first real node in the list. Throws an error if the list is empty.
ListElement List::front() const{
    if(this->length() <= 0){
        throw std::length_error("front() pre-condition: length() > 0");
    }
    return frontDummy->next->data;
}

// Returns the data of the last real node in the list. Throws an error if the list is empty.
ListElement List::back() const{
    if(this->length() <= 0){
        throw std::length_error("back() pre-condition: length() > 0");
    }
    return backDummy->prev->data;
}

// Returns the data of the node immediately after the cursor. Throws an error if the cursor is at the end of the list.
ListElement List::peekNext() const{
    if(this->position() >= this->length()){
        throw std::range_error("peekNext() pre-condition: position() < length()");
    }
    return afterCursor->data;
}

// Returns the data of the node immediately before the cursor. Throws an error if the cursor is at the start of the list.
ListElement List::peekPrev() const{
    if(this->position() <= 0){
        throw std::range_error("peekPrev() pre-condition: position() > 0");
    }
    return beforeCursor->data;
}

// Returns the current position of the cursor in the list.
int List::position() const{
    return this->pos_cursor;
}

// Moves the cursor to the front of the list (before the first real element).
void List::moveFront(){
    beforeCursor = frontDummy;
    afterCursor = beforeCursor->next;
    this->pos_cursor = 0;
}

// Moves the cursor to the back of the list (after the last real element).
void List::moveBack(){
    afterCursor = backDummy;
    beforeCursor = afterCursor->prev;
    this->pos_cursor = this->length();
}

// Moves the cursor one position forward in the list and returns the data of the node now before the cursor.
int List::moveNext(){
    if(this->position() >= this->length()){
        throw std::range_error("moveNext() pre-condition: position()<length()");
    }
    beforeCursor = beforeCursor->next;
    afterCursor = afterCursor->next;
    this->pos_cursor ++;
    return beforeCursor->data;
}

// Moves the cursor one position backward in the list and returns the data of the node now after the cursor.
int List::movePrev(){
    if(this->position() <= 0){
        throw std::range_error("movePrev() pre-condition: position()>0");
    }
    beforeCursor = beforeCursor->prev;
    afterCursor = afterCursor->prev;
    this->pos_cursor --;


    return afterCursor->data;
}

// Inserts a new node with given data immediately after the cursor.
void List::insertAfter(int x){
    Node* N = new Node(x);
    N->prev = beforeCursor;
    N->next = afterCursor;
    beforeCursor->next = N;
    afterCursor->prev = N;
    afterCursor = N;
    this->num_elements ++;
}

// Inserts a new node with given data immediately before the cursor.
void List::insertBefore(int x){
    Node* N = new Node(x);
    N->prev = beforeCursor;
    N->next = afterCursor;
    beforeCursor->next = N;
    afterCursor->prev = N;
    beforeCursor = N;
    this->num_elements ++;
    this->pos_cursor ++;
}

// Replaces the data of the node immediately after the cursor with x. Throws an error if the cursor is at the end.
void List::setAfter(ListElement x){
  if(this->position() >= this->length()){
        throw std::range_error("setAfter() pre-condition: pre: position()<length()");
        return;
    }
  this->afterCursor->data = x;
}

// Replaces the data of the node immediately before the cursor with x. Throws an error if the cursor is at the start.
void List::setBefore(ListElement x){
  if(this->position() <=0){
        throw std::range_error("setBefore() pre-condition: pre: position()>0");
        return;
    }
  this->beforeCursor->data = x;
}

// Deletes the node immediately after the cursor. Throws an error if the cursor is at the end.
void List::eraseAfter(){
    if(this->position() >= this->length()){
        throw std::range_error("eraseAfter() pre-condition: pre: position()<length()");
        return;
    }
    Node* erase = afterCursor;
    afterCursor = afterCursor->next;
    beforeCursor->next = afterCursor;
    afterCursor->prev = beforeCursor;
    this->num_elements --;
    delete erase;
}

// Deletes the node immediately before the cursor. Throws an error if the cursor is at the start.
void List::eraseBefore(){
    if(this->position() <= 0){
        throw std::range_error("eraseBefore() pre-condition: pre: position()>0");
        return;
    }


    Node* erase = beforeCursor;
    beforeCursor = beforeCursor->prev;
    beforeCursor->next = afterCursor;
    afterCursor->prev = beforeCursor;
    this->pos_cursor --;
    this->num_elements --;
    delete erase;
}

// Searches for the next occurrence of x in the list after the cursor, moves the cursor to that element, and returns the position. Returns -1 if not found.
int List::findNext(int x){
    Node* N = this->afterCursor;
    int r = 1;
    while(N != this->backDummy){
        if(N->data == x){
            this->beforeCursor = N;
            this->afterCursor = N->next;
            this->pos_cursor += r;
            return this->pos_cursor;
        }
        N = N->next;
        r ++;
    }
    // If not found, move cursor to the end.
    this->afterCursor = backDummy;
    this->beforeCursor = afterCursor->prev;
    this->pos_cursor = length();
    return -1;
}

// Searches for the previous occurrence of x in the list before the cursor, moves the cursor to that element, and returns the position. Returns -1 if not found.
int List::findPrev(int x){
    Node* N = this->beforeCursor;
    int l = 1;
    while(N != this->frontDummy){
        if(N->data == x){
            beforeCursor = N->prev;
            afterCursor = N;
            pos_cursor -= l;
            return this->pos_cursor;
        }
        N = N->prev;
        l ++;
    }
    // If not found, reset cursor to the start.
    this->beforeCursor = frontDummy;
    this->afterCursor = beforeCursor->next;
    this->pos_cursor = 0;
    return -1;
}

// Removes duplicate elements in the list.
void List::cleanup(){
    Node* N = this->frontDummy->next;
    int outter_pos = 0;
    while(N != this->backDummy){
        int i = N->data;
        Node* M = N->next;
        int inner_pos = outter_pos + 1;
        while(M != this->backDummy){
            int j = M->data;
            Node* before = M->prev;
            Node* after = M->next;
            Node* erase = M;
            M = after;
            if(j == i){
                before->next = after;
                after->prev = before;
                delete erase;
                this->num_elements --;
                if(inner_pos <= pos_cursor){
                    pos_cursor --;
                }
            }
            inner_pos ++;
        }
        outter_pos ++;
        N = N->next;
    }
    // Reset cursor position after cleanup.
    int pos = this->pos_cursor;
    this->moveFront();
    while(this->position() != pos){
        this->moveNext();
    }
}

// Clears all elements from the list, leaving it empty.
void List::clear(){
    this->moveFront();
    while(this->length() > 0){
        this->eraseAfter();
    }
}

// Concatenates the current list with another list L and returns the new list.
List List::concat(const List& L) const{
    List newL;
    newL.moveBack();
    Node* N;

    // Copy elements from the current list.
    N = this->frontDummy->next;
    while(N != this->backDummy){
        newL.insertBefore(N->data);
        N = N->next;
    }

    // Copy elements from list L.
    N = L.frontDummy->next;
    while(N != L.backDummy){
        newL.insertBefore(N->data);
        N = N->next;
    }
    // Reset cursor position in the new list.
    newL.beforeCursor = newL.frontDummy;


    newL.afterCursor = newL.beforeCursor->next;
    newL.pos_cursor = 0;
    return newL;
}

// Returns a string representation of the list.
std::string List::to_string() const{
    string s;
    s += "(" ;
    List::Node* N = this->frontDummy->next;
    while(N != this->backDummy){
        s += std::to_string(N->data);
        if(N->next != this->backDummy){
            s +=", " ;
        }
        N = N->next;
    }
    s += ")" ;
    return s;
}

// Compares two lists for equality.
bool List::equals(const List& R) const{
    if(this->length() != R.length()){
        return false;
    }
    Node* N = this->frontDummy->next;


    Node* M = R.frontDummy->next;
    while(N != this->backDummy && M != R.backDummy){
        int n = N->data;
        int m = M->data;
        if(n != m){
            return false;
        }
        N = N->next;
        M = M->next;
    }
    return true;
}

// Overloads the << operator to print the list using ostream.
std::ostream& operator<<(std::ostream& stream, const List & L ){
    return stream << L.to_string();
}

// Overloads the == operator to compare two lists.
bool operator==(const List& A, const List& B ){
    return A.equals(B);
}

// Overloads the assignment operator to copy a list.
List& List::operator=( const List& L ){
    if( this != &L ){
        // Use the copy-swap idiom for assignment.
        List temp = L;
        std::swap(frontDummy, temp.frontDummy);
        std::swap(backDummy, temp.backDummy);
        std::swap(beforeCursor, temp.beforeCursor);
        std::swap(afterCursor, temp.afterCursor);
        std::swap(pos_cursor, temp.pos_cursor);
        std::swap(num_elements, temp.num_elements);
    }
    return *this;
}

#include<iostream>
#include<fstream>
#include<string>
#include <sstream>
#include"List.h"


using namespace std;

// prototype function for shuffle
void shuffle(List& D);


int main(int argc, char * argv[]){
    // input and output file stream objects
    ifstream in;
    ofstream out;
    stringstream ss; // string stream for converting command line argument to integer

    // checks if correct # of command line arguments is provided
    if( argc != 2 ){
        cerr << "Usage: " << argv[0] << " <size of deck>" << endl;
        return(EXIT_FAILURE);
    }
    // displays header for the output
    cout << "size of deck       shuffle count" << endl;
    cout << "------------------------------" << endl;
    
    // reads the size of the deck from command line argument
    int deck;
    ss << argv[1];
    ss >> deck;

    List L; // creates List L
    // populates list with values from 1 to deck
    for(int i = 1; i <= deck; i++){
        L.insertBefore(i-1);
        List D(L);

        int count = 0;
        do{
            shuffle(D);
            count ++;
        }while(! D.equals(L));

        // Print the sullfle count
        cout << i << "\t\t" << count << endl;
    }
    in.close();     // Close file stream
    out.close();    // Close file stream

    return(EXIT_SUCCESS);
}

// shuffles List D
void shuffle(List& D){
    List temp; // temp List to store first half of List D
    int split = D.length() / 2;
    D.moveFront();

    // moves first half of List D to temp List
    for(int i = 0; i < split; i++){
        int data = D.moveNext();
        temp.insertBefore(data);
        D.eraseBefore();
    }

    // resets positions of temp List and List D
    temp.moveFront();
    while(D.position() < D.length()){
        D.moveNext();

        // if elements in temp List, inserts them into List D
        if(temp.position() < temp.length()){
            D.insertAfter(temp.moveNext());
            D.moveNext();
        }

    }
}

#include<iostream>
#include<fstream>
#include<string>
#include <sstream>
#include "List.h"


using namespace std;




void shuffle(List& D);


int main(int argc, char * argv[]){

    ifstream in;
    ofstream out;
    stringstream ss;

   
    // Check command line for correct number of arguments
    if( argc != 2 ){
        cerr << "Usage: " << argv[0] << " <deck size>" << endl;
        return(EXIT_FAILURE);
    }
    cout << "deck size       shuffle count" << endl;
    cout << "------------------------------" << endl;
    
    int deck;
    ss << argv[1];
    ss >> deck;

    List L;
    for(int i = 1; i <= deck; i++){
        
        L.insertBefore(i-1);
        List D(L);

        int count = 0;
        do{
            shuffle(D);
            count ++;
        }while(! D.equals(L));

        // Print the sullfle count
        cout << i << "\t\t" << count << endl;
    }
    in.close();
    out.close();

    return(EXIT_SUCCESS);
}

void shuffle(List& D){
    
    List temp;
    int split = D.length() / 2;
    D.moveFront();
    for(int i = 0; i < split; i++){
        int data = D.moveNext();
        temp.insertBefore(data);
        D.eraseBefore();
    }

    temp.moveFront();
    while(D.position() < D.length()){
        D.moveNext();
        if(temp.position() < temp.length()){
            D.insertAfter(temp.moveNext());
            D.moveNext();
        }

    }

}

//perfect shuffle
    //three decks: 
    //1 starting state of d
    // 2 temps L (left) and R (right) (which you need to delete at end of shuffle function)
    // shuffle 
void shuffle(List& D){
    List L;
    List R;
    int x;
    int y;


    int n = D.length();
    
    // split D into the L and R halves
    D.moveFront();
    while (D.position() < n / 2) {
        x = D.moveNext();
        L.insertBefore(x);
    }   
    while (D.position() < n) {
        y = D.moveNext();
        R.insertBefore(y);
    }
    // clear D and shuffle L and R back into D
    D.clear();
    L.moveFront();


    R.moveFront();
    D.moveFront();
    while (L.position() < L.length() && R.position() < R.length()) {
        x = R.moveNext();
        D.insertBefore(x);

        y = L.moveNext();
        D.insertBefore(y);
    }

    //edge case
    while (L.position() < L.length()) {
    D.insertBefore(L.moveNext());
    }

    while (R.position() < R.length()) {
        D.insertBefore(R.moveNext());
    }


    //erase L and R
    L.clear();
    R.clear();
}

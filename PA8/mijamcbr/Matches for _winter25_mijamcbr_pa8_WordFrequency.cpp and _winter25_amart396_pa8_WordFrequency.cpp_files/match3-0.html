#include <algorithm>
#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <vector>
#include "Dictionary.h"

using namespace std;

int main (int argc, char** argv) {
    ifstream IN;


    ofstream OUT;
    string line;
    if (argc != 3)
    {
        cerr << "Usage: <input file> <output file>\n";
        return (EXIT_FAILURE);
    }
    // open files for reading and writing 
    IN.open(argv[1]);
    if( !IN.is_open() ){
        cerr << "Unable to open file for reading\n";
        return(EXIT_FAILURE);
    }
    OUT.open(argv[2]);
    if( !OUT.is_open() ){
        cerr << "Unable to open file or writing\n";
        return(EXIT_FAILURE);
    }
    Dictionary D;
    while (getline(IN, line)) {
        istringstream iss (line);
        string delim = " \t\\\"\',<.>/?;:[{]}|`~!@#$%^&*()-_=+0123456789";
        string word;
        size_t start = min(line.find_first_not_of(delim, 0), line.length());
        size_t end = min(line.find_first_of(delim, start), line.length());
        word = line.substr(start, end - start);
        while (word != "") {
            transform(word.begin(), word.end(), word.begin(), ::tolower);
            if (D.contains(word)) {
                D.setValue(word, D.getValue(word) + 1);
            } else {
                D.setValue(word, 1);
            }
            start = min(line.find_first_not_of(delim, end+1), line.length());
            end = min(line.find_first_of(delim, start), line.length());
            word = line.substr(start, end - start);
        }
    } 
    OUT << endl << D.to_string() << endl;
    IN.close();
    OUT.close();
    return 0;

}

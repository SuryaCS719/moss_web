using namespace std;
int main(int argc, char *argv[]) {
    ifstream inputFile1;


    ofstream inputFile2;
    string currLine;
    //Heavily templated from Claude AI
    if (argc != 3)
    {
        cerr << "Usage: input file> <output file>\n";
        return (EXIT_FAILURE);
    }
    
    //Now open files for reading and writing
    inputFile1.open(argv[1]);
    if(!inputFile1.is_open() ){ //check error
        cerr << "Unable to open file for reading\n";
        return(EXIT_FAILURE); }
    inputFile2.open(argv[2]);
    if( !inputFile2.is_open() ){ //check error
        cerr << "Unable to open file or writing\n";
        return(EXIT_FAILURE); }
        
    //Now begin checking Frequency!!
    Dictionary F;
    while (getline(inputFile1, currLine)) {
        istringstream iss (currLine);
        //format from given functoin in p8
        string delim = " \t\\\"\',<.>/?;:[{]}|`~!@#$%^&*()-_=+0123456789";
        string word;
        size_t start = min(currLine.find_first_not_of(delim, 0), currLine.length());
        size_t end = min(currLine.find_first_of(delim, start), currLine.length());
        word = currLine.substr(start, end - start);
        
        while (word != "") {
            transform(word.begin(), word.end(), word.begin(), ::tolower);
            if (F.contains(word)) {
                F.setValue(word, F.getValue(word) + 1);
            } else {
                F.setValue(word, 1);
            }
            start = min(currLine.find_first_not_of(delim, end+1), currLine.length());
            end = min(currLine.find_first_of(delim, start), currLine.length());
            word = currLine.substr(start, end - start); 
        }
    } 
    inputFile2 << endl << F.to_string() << endl;
    inputFile2.close();
    inputFile1.close();
    
    return 0;
}

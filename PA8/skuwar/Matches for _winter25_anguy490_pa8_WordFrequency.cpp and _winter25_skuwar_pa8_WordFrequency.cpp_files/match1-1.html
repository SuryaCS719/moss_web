#include "Dictionary.h"
#include <iostream>
#include <fstream>
#include <string>
#include <algorithm>

// Function to convert a string to lowercase


std::string toLowercase(const std::string& str) {
    std::string lowerStr = str;
    std::transform(lowerStr.begin(), lowerStr.end(), lowerStr.begin(), ::tolower);
    return lowerStr;
}

int main(int argc, char* argv[]) {
    // Check for correct number of arguments
    if (argc != 3) {
        std::cerr << "Usage: " << argv[0] << " <input file> <output file>" << std::endl;
        return 1;
    }

    // Open input file
    std::ifstream inFile(argv[1]);
    if (!inFile.is_open()) {
        std::cerr << "Unable to open file " << argv[1] << " for reading" << std::endl;
        return 1;
    }

    // Create a Dictionary to store word frequencies
    Dictionary wordFreq;

    // Define the delimiter for tokenizing the input file
    std::string delim = " \t\\\"\',<.>/?;:[{]}|`~!@#$%^&*()-_=+0123456789";

    // Read the input file line by line


    std::string line;
    while (std::getline(inFile, line)) {
        size_t start = line.find_first_not_of(delim, 0);
        size_t end = line.find_first_of(delim, start);

        while (start != std::string::npos || end != std::string::npos) {
            std::string word = line.substr(start, end - start);
            word = toLowercase(word);

            // If the word is already in the dictionary, increment its frequency
            if (wordFreq.contains(word)) {
                wordFreq.getValue(word)++;
            } else {
                // Otherwise, add the word to the dictionary with a frequency of 1
                wordFreq.setValue(word, 1);
            }

            start = line.find_first_not_of(delim, end);
            end = line.find_first_of(delim, start);
        }
    }

    // Close the input file
    inFile.close();

    // Open output file
    std::ofstream outFile(argv[2]);


    if (!outFile.is_open()) {
        std::cerr << "Unable to open file " << argv[2] << " for writing" << std::endl;
        return 1;
    }

    // Write the word frequencies to the output file
    outFile << wordFreq.to_string();

    // Close the output file
    outFile.close();

    return 0;
}

#include <iostream>
#include <fstream>
#include <string>
#include <cctype>
#include "Dictionary.h"



using namespace std;

// Helper function to convert a string to lowercase
string toLowercase(const string& word) {
    string result;
    for (char ch : word) {
        result += tolower(ch);
    }
    return result;
}

int main(int argc, char* argv[]) {
    // Check for correct number of arguments
    if (argc != 3) {
        cerr << "Usage: " << argv[0] << " <input file> <output file>" << endl;
        return EXIT_FAILURE;
    }

    ifstream inFile(argv[1]);
    ofstream outFile(argv[2]);

    // Check if input file opened successfully
    if (!inFile.is_open()) {
        cerr << "Unable to open input file: " << argv[1] << endl;
        return EXIT_FAILURE;
    }

    // Check if output file opened successfully
    if (!outFile.is_open()) {
        cerr << "Unable to open output file: " << argv[2] << endl;
        return EXIT_FAILURE;
    }

    // Create an empty Dictionary
    Dictionary wordFreq;

    string line;
    string delim = " \t\\\"\',<.>/?;:[{]}|`~!@#$%^&*()-_=+0123456789";

    // Read the input file line by line
    while (getline(inFile, line)) {
        size_t begin = line.find_first_not_of(delim);
        while (begin != string::npos) {
            size_t end = line.find_first_of(delim, begin);
            string word = line.substr(begin, end - begin);
            word = toLowercase(word);  // Convert word to lowercase

            if (word != "") {
                // If word is already in dictionary, increment its frequency
                if (wordFreq.contains(word)) {
                    wordFreq.getValue(word) += 1;
                } else {
                    // If word is new, insert it with a frequency of 1
                    wordFreq.setValue(word, 1);
                }
            }

            // Move to next word
            begin = line.find_first_not_of(delim, end);
        }
    }

    // Output the dictionary to the output file
    outFile << wordFreq.to_string();

    // Clean up and close files
    inFile.close();
    outFile.close();

    return EXIT_SUCCESS;
}

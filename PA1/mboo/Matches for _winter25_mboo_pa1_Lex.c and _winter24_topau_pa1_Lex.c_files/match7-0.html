#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
#include"List.h"
#include<string.h>
#define MAX_LEN 100



char *strdup(const char *str){
   int n = strlen(str) + 1;
   char *dup = malloc(n);
   if(dup){
      strcpy(dup, str); 
   }
   return dup;
}

int main(int argc, char * argv[]){
   int line_count;
   FILE *in, *out;
   char line[MAX_LEN];

   if( argc != 3 ){
      fprintf(stderr, "Usage: %s <input file> <output file>\n", argv[0]);
      exit(1);
   }

   in = fopen(argv[1], "r");
   if( in==NULL ){
      fprintf(stderr, "Unable to open file %s for reading\n", argv[1]);
      exit(1);
   }



   out = fopen(argv[2], "w");
   if( out==NULL ){
      fprintf(stderr, "Unable to open file %s for writing\n", argv[2]);
      exit(1);
   }

   line_count = 0;
   while( fgets(line, MAX_LEN, in) != NULL )  {
      line_count++;
   }

   rewind(in);
   char** str_list = (char**) calloc(sizeof(char*), line_count + 1);
   int str_list_index = 0;
   while( fgets(line, MAX_LEN, in) != NULL )  {
      str_list[str_list_index] = strdup(line); 
      str_list_index++;
   }

   List L = newList();
   append(L, 0);
   moveFront(L);
   for(int i = 1; i < line_count; i++){ 
      moveFront(L);
      while(index(L) >= 0){
         if(strcmp(str_list[get(L)], str_list[i]) > 0){
            break;
         }
         moveNext(L);
      }
      if(index(L) < 0){
         append(L, i);
      }else{
         insertBefore(L,i);
      }
   }
 
   moveFront(L);
   while(index(L) >= 0){
      int x = get(L);
      fprintf(out, "%s", str_list[x]);
      moveNext(L);
   }

   freeList(&L); 
   for(int i = 0; i < line_count + 1; i++){
      free(str_list[i]);
   }
   free(str_list);
   fclose(in);
   fclose(out);

   return(0);
}

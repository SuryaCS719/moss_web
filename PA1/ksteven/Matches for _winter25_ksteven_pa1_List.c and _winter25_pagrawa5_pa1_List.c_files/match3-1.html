#include<stdio.h>

typedef struct NodeObj *Node;

typedef struct NodeObj{
  int data;
  Node next;
  Node prev;
} NodeObj;

typedef struct ListObj{
  Node front;
  Node back;
  Node cursor;
  int length;
  int index;
} ListObj;

Node newNode(int data){
   Node node = malloc(sizeof(NodeObj));
   assert(node!=NULL);
   node->data = data;
   node->next = NULL;
   node->prev = NULL;
   return(node);
}


void freeNode(Node* pN){


   if(pN!=NULL && *pN!=NULL){
      free(*pN);
      *pN = NULL;
   }
}

List newList(void){
  List L = malloc(sizeof(ListObj));
  L->front = NULL;
  L->back = NULL;
  L->cursor = NULL;
  L->index = -1;
  L->length = 0;
  return(L);
}

void freeList(List* pL){
  if(pL != NULL && *pL != NULL){
    while(!empty(*pL)){
      deleteFront(*pL);
    }
    free(*pL);
    *pL = NULL;
  }
}

bool equals(List A, List B){
  if(A == NULL || B == NULL){
    printf("[ERR]: cannot call equals() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  bool eq;
  Node node1, node2;

  eq = (A->length == B->length);
  node1 = A->front;
  node2 = B->front;
  while(eq && node1!=NULL){
    eq = (node1->data==node2->data);


    node1 = node1->next;
    node2 = node2->next;
  }
   return eq;
}

void clear(List L){
  if(L == NULL){
    printf("[ERR]: cannot call clear() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  while(L->length > 0){
    deleteFront(L);
  }
  L->front = NULL;
  L->back = NULL;
  L->cursor = NULL;
  L->index = -1;
  L->length = 0;
}

void set(List L, int x){
  if(empty(L)){
    printf("[ERR]: cannot call set() on an empty List\n");
  exit(EXIT_FAILURE);
  }
  
  if(L->index >= 0 && L->length > 0){
    L->cursor->data = x;
  }
}

void moveFront(List L){
  if(L->length != 0){
    L->cursor = L->front;
    L->index = 0;
  }
}

void moveBack(List L){
  if(L==NULL){
    printf("[ERR]: cannot call moveBack() on a NULL List\n");
    exit(EXIT_FAILURE);
  }

  if(L->length==0){
    printf("[ERR]: cannot call moveBack() on an empty List\n");
    exit(EXIT_FAILURE);
  }

  if(L->length > 0){
    L->cursor = L->back;
    L->index = L->length-1;
  }
}

void movePrev(List L){
  if(L==NULL){
    printf("[ERR]: cannot call movePrev() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length==0){
    printf("[ERR]: cannot call movePrev() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  if(L->cursor != NULL){
    if(L->cursor != L->front){
      L->cursor = L->cursor->prev;
      L->index--;
    } 
    else if(L->cursor == L->front){
      L->cursor = NULL;
      L->index = -1;
    }
  }
}

void moveNext(List L){
  if(L==NULL){
    printf("[ERR]: cannot call moveNext() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length==0){
    printf("[ERR]: cannot call moveNext() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  
  if(L->cursor != NULL && L->cursor != L->back){
    L->cursor = L->cursor->next;
    L->index++;
  } 


  else if(L->cursor != NULL && L->cursor == L->back){
    L->cursor = NULL;
    L->index = -1;
  }
}

void prepend(List L, int x){
  if(L==NULL){
    printf("[ERR]: cannot call prepend() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  Node node = newNode(x);
  if(length(L) == 0){
    L->front = L->back = node;
  } 
  else{
    L->front->prev = node;
    node->next = L->front;
    L->front = node;
    if(L->cursor == L->front){
      L->index++;
    } 
    else if(L->index != -1){
        L->index++;
    }
  }
  L->length++;
}

void append(List L, int x){
  if(L==NULL){
    printf("[ERR]: cannot call append() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  Node node = newNode(x);
  if(L->back != NULL){
    node->prev = L->back; 
    L->back->next = node;
    node->prev = L->back;
  } 
  else{
    L->front = node;
  }
  L->length++;
  L->back = node;
}

void insertBefore(List L, int x){
  if(L==NULL){
    printf("[ERR]: cannot call insertBefore() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length==0){
    printf("[ERR]: cannot call insertBefore() on an empty List\n");

    exit(EXIT_FAILURE);
  }
  if(L->index >= 0 && L->length > 0){
    Node node = newNode(x);
    node->next = L->cursor;
    if(L->cursor->prev != NULL){
      node->prev = L->cursor->prev;
      node->prev->next = node;
    } 
    else{
        L->front = node;
    }
    L->cursor->prev = node;
    L->length++;
    L->index++;
  }
}

void deleteFront(List L){
  if(L==NULL){
    printf("[ERR]: cannot call deleteFront() on a NULL List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length==0){
    printf("[ERR]: cannot call deleteFront() on an empty List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length<=0){
    printf("[ERR]: cannot call deleteFront() on a List that's out of range\n");
    exit(EXIT_FAILURE);
  }
  if(L->length > 0){
    Node node = L->front->next;
    freeNode(&(L->front));
    L->front = node;
    if (L->cursor == L->front){
      L->back = NULL;
    }
    L->length--;
    L->index--;
  }
}

void deleteBack(List L){
  if(L==NULL){
    printf("List Error: calling deleteBack() on NULL List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length==0){
    printf("List Error: calling deleteBack() on empty List\n");
    exit(EXIT_FAILURE);
  }
  if(L->length<=0){
    printf("List Error: calling deleteBack() on out of range List\n");
    exit(EXIT_FAILURE);
  }
  if(L->cursor == L->back){
    L->cursor = NULL;
    L->index = -1;
  }
  Node node = L->back;
  L->back = L->back->prev;
  if(L->back != NULL){
    L->back->next = NULL;
  } 
  else{
    L->front = NULL;
  }
  freeNode(&node);
  L->length--;
}

void delete(List L){
  if(L->cursor == L->front){
    deleteFront(L);
  } 
  else if(L->cursor == L->back){
    deleteBack(L);
  } 
  else if(L->index >= 0 && L->length > 0){
    L->cursor->next->prev = L->cursor->prev;
    L->cursor->prev->next = L->cursor->next;
    Node node = L->cursor;
    L->cursor = NULL;
    L->index = -1;
    L->length -= 1;
    freeNode(&node);
  }
}

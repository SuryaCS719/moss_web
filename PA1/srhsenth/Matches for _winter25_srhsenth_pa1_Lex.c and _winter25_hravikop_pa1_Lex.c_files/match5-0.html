#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "List.h"

#define MAX_LEN 255



int main(int argc, char* argv[]) {
    char str[MAX_LEN];
    List strList = newList();

    if (argc != 3) {
        fprintf(stderr, "Need exactly three arguments.");
        exit(EXIT_FAILURE);
    }

    FILE* in = fopen(argv[1], "r");
    if (in == NULL) {
        fprintf(stderr, "File read error, %s.", argv[1]);
        exit(EXIT_FAILURE);
    }



    FILE* out = fopen(argv[2], "w");
    if (out == NULL) {
        fprintf(stderr, "Error when writing file, %s.", argv[2]);
        exit(EXIT_FAILURE);
    }
   
   //simple loop to count lines 
    int lines = 0;
    while (fgets(str, MAX_LEN, in) != NULL) {
        lines++;
    }

    fclose(in);
    in = fopen(argv[1], "r");

    char strArr[lines][MAX_LEN];
    lines = 0;  // reset lines
    while (fgets(str, MAX_LEN, in) != NULL) {
        for (int i = 0; i < MAX_LEN; i++) {
            strArr[lines][i] = str[i];
        }
        lines++;
    }
   
   //implementing insertion sort algorithm 
    for (int i = 0; i < lines; i++) {
        moveBack(strList);
        while (index(strList) > -1 &&  strcmp(strArr[i], strArr[get(strList)]) < 0) {
            movePrev(strList);
        }
        if (index(strList) > -1) {
            insertAfter(strList, i);
        }
        else {
            prepend(strList, i);
        }
    }

    moveFront(strList);
    while (index(strList) > -1) {
        fprintf(out, "%s", strArr[get(strList)]);
        moveNext(strList);
    }

    freeList(&strList);
    strList = NULL;

    fclose(in);
    fclose(out);

    return 0;
}
